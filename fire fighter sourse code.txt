#define BLYNK_TEMPLATE_NAME "Fire Fighter Navigator"
#define BLYNK_AUTH_TOKEN "imsxzozHbIqWalNSYkoCxOpEhvZeHgrp"

#include <WiFi.h>
#include <BlynkSimpleEsp32.h>
#include <Wire.h>
#include <DHT.h>
#include <MPU6050.h>
#include <HardwareSerial.h>
#include <TinyGPS++.h>
#include "MAX30100_PulseOximeter.h"

// --- WiFi ---
char auth[] = BLYNK_AUTH_TOKEN;
char ssid[] = "Mahii";
char pass[] = "Mahii.zz";

// --- MQ2 Gas Sensor ---
#define MQ2_PIN 34
#define RED_LED_PIN 32
#define GREEN_LED_PIN 33
#define GAS_THRESHOLD 500

// --- DHT11 ---
#define DHTPIN 4
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

// --- MPU6050 ---
MPU6050 mpu;

// --- MAX30100 ---
PulseOximeter pox;
uint32_t lastHRUpdate = 0;

// --- GPS (Neo-6M) ---
HardwareSerial GPSserial(1);
TinyGPSPlus gps;

// --- Setup ---
void setup() {
  Serial.begin(115200);

  pinMode(RED_LED_PIN, OUTPUT);
  pinMode(GREEN_LED_PIN, OUTPUT);
  digitalWrite(RED_LED_PIN, LOW);
  digitalWrite(GREEN_LED_PIN, HIGH);

  dht.begin();

  Wire.begin(21, 22);  // SDA=21, SCL=22
  mpu.initialize();

  // GPS UART
  GPSserial.begin(9600, SERIAL_8N1, 16, -1); // RX=16, no TX

  // MAX30100
  Serial.print("Initializing MAX30100... ");
  if (!pox.begin()) {
    Serial.println("FAILED");
  } else {
    Serial.println("SUCCESS");
  }

  Blynk.begin(auth, ssid, pass);
  Serial.println("🚀 Firefighter Suit Started...");
}

// --- Loop ---
void loop() {
  Blynk.run();

  /********** MQ2 **********/
  int gasValue = analogRead(MQ2_PIN);
  Blynk.virtualWrite(V2, gasValue);

  if (gasValue > GAS_THRESHOLD) {
    digitalWrite(RED_LED_PIN, HIGH);
    digitalWrite(GREEN_LED_PIN, LOW);
    Blynk.virtualWrite(V7, 1);
  } else {
    digitalWrite(RED_LED_PIN, LOW);
    digitalWrite(GREEN_LED_PIN, HIGH);
    Blynk.virtualWrite(V7, 0);
  }

  /********** DHT11 **********/
  float temp = dht.readTemperature();
  float hum  = dht.readHumidity();

  if (!isnan(temp) && !isnan(hum)) {
    Blynk.virtualWrite(V0, temp);
    Blynk.virtualWrite(V1, hum);
  }

  /********** MPU6050 **********/
  int16_t ax, ay, az;
  int16_t gx, gy, gz;
  mpu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);

  Blynk.virtualWrite(V10, ax / 16384.0);
  Blynk.virtualWrite(V11, ay / 16384.0);
  Blynk.virtualWrite(V12, az / 16384.0);
  Blynk.virtualWrite(V13, gx / 131.0);
  Blynk.virtualWrite(V14, gy / 131.0);
  Blynk.virtualWrite(V15, gz / 131.0);

  /********** MAX30100 **********/
  pox.update();
  if (millis() - lastHRUpdate > 1000) {
    float hr = pox.getHeartRate();
    float spo2 = pox.getSpO2();
    Blynk.virtualWrite(V20, hr);
    Blynk.virtualWrite(V21, spo2);

    // Danger check
    String danger = "Safe";
    if (spo2 < 90) danger = "Low SpO2";
    if (hr < 50 || hr > 120) danger = "Abnormal HR";
    Blynk.virtualWrite(V22, danger);

    lastHRUpdate = millis();
  }

  /********** GPS **********/
  while (GPSserial.available()) {
    if (gps.encode(GPSserial.read())) {
      if (gps.location.isValid()) {
        Blynk.virtualWrite(V30, gps.location.lat(), 6);
        Blynk.virtualWrite(V31, gps.location.lng(), 6);
        Blynk.virtualWrite(V32, gps.speed.kmph());
        Blynk.virtualWrite(V33, gps.altitude.meters());

        String mapLink = "https://maps.google.com/?q=" +
                         String(gps.location.lat(), 6) + "," +
                         String(gps.location.lng(), 6);
        Blynk.virtualWrite(V34, mapLink);
      }
    }
  }

  delay(500);
}
